181 auto CSP<ValueType>::consistent(ValueType const& value, usize const& varId) const -> bool {
182 usize i, end = variables.size();
183 for (i = 0; i < varId; ++i) {
184 if (!variables[i].unassigned) {
185 ValueType const& value_i = variables[i].value;
186 if (!checkBinaryConstraint(value_i, i, value, varId) ||
187 !checkBinaryConstraint(value, varId, value_i, i)) {
188 return false;
189 }
190 }
191 }
192 for (i = varId + 1; i < end; ++i) {
193 if (!variables[i].unassigned) {
194 ValueType const& value_i = variables[i].value;
195 if (!checkBinaryConstraint(value_i, i, value, varId) ||
196 !checkBinaryConstraint(value, varId, value_i, i)) {
197 return false;

201 return true;

215 template <typename ValueType>
216 auto CSP<ValueType>::revise(usize const& varId1, usize const& varId2) -> bool {
217 bool revised = false;
218 vector<ValueType>& D1 = variables[varId1].domain;
219 auto it = D1.begin();
220 while (it != D1.end()) {
221 if (satisfy(*it, varId1, varId2)) {
222 ++it;
223 } else {
224 it = D1.erase(it);
225 revised = true;
226 }
227 }
228 return revised;

231 template <typename ValueType>
232 auto CSP<ValueType>::ac3() -> bool {
233 queue<Arc> arcQueue = arcs;
234 while (!arcQueue.empty()) {
235 Arc& currentArc = arcQueue.front();
236 usize& i = currentArc.first;
237 usize& j = currentArc.second;
238 // 维护Xi对Xj的弧相容
239 if (revise(i, j)) {
240 // Xi的值域发生了更改
241 if (variables[i].domain.empty()) {
242 return false;
243 }
244 // 将以Xi为tail的弧入队
245 for (usize k = 0; k < variables.size(); ++k) {
246 if (binaryConstraints[k][i] && k != i) {
247 arcQueue.emplace(k, i);
248 }
249 }
250 }
251
252 arcQueue.pop();
253 }
254 return true;
255 }

271 auto CSP<ValueType>::isSolution() -> bool {
272 for (Var& var : variables) {
273 if (var.unassigned) {
274 return false;
275 }
276 }
277 for (usize i = 0; i < variables.size(); ++i) {
278 for (usize j = 0; j < variables.size(); ++j) {
279 if (j == i) {
280 continue;
281 }
282 if (!checkBinaryConstraint(variables[i].value, i, variables[j].value, j) ||
283 !checkBinaryConstraint(variables[j].value, j, variables[i].value, i)) {
284 return false;
285 }
286 }
287 }
288 return true;

317 auto CSP<ValueType>::backtrack() -> void {
318 if (isSolution()) {
319 return;
320 }
321 usize varId = selectUnassignedVariableId();
322 vector<ValueType> orderDomain = orderDomainValues(varId);
323 for (ValueType& value : orderDomain) {

325 auto rcd = record();

328 backtrack();

365 usize i = 0;
366 vector<usize> conflictVariablesId;
367 for (Var const& var : variables) {
368 if (!var.unassigned && getConflictsNum(var.value, i) > 0) {
369 conflictVariablesId.push_back(i);

371 ++i;

374 return 0;

384 auto CSP<ValueType>::minConflictsNumValue(usize const& varId) -> ValueType {
385 vector<ValueType> minConflictsValues;
386 usize curConflictsNum, minConflictsNum = variables.size();
387 for (ValueType const& val : variables[varId].domain) {
388 curConflictsNum = getConflictsNum(val, varId);
389 if (curConflictsNum < minConflictsNum) {
390 minConflictsValues.clear();
391 minConflictsValues.push_back(val);
392 minConflictsNum = curConflictsNum;
393 } else if (curConflictsNum == minConflictsNum) {
394 minConflictsValues.push_back(val);
395 }
396 }
397 if (minConflictsValues.size() == 1) {
398 return minConflictsValues[0];
399 } else {
400 random_device rd;
401 mt19937 gen(rd());
402 uniform_int_distribution<usize> dis(0, minConflictsValues.size() - 1);
403 return minConflictsValues[dis(gen)];
404 }
