240 template <CriterionType Criterion = INFO_GAIN, bool PrintInfo = false>
241 static auto importance(
242 vector<Attribute*> const& attributes,
243 Attribute* const& target,
244 vector<pair<Example*, f64>>& examples,
245 pair<string, f64>* nodeInfo = nullptr
246 ) -> Attribute* {
247
248 function<f64(Attribute* const&, Attribute* const&, vector<pair<Example*, f64>>&)> func;
249 function<bool(f64 const&, f64 const&)> firstBetter;
250 if constexpr (Criterion == CriterionType::INFO_GAIN) {
251 func = infoGain;
252 firstBetter = greater(); // cur(first) > select(second)
253 } else if constexpr (Criterion == CriterionType::INFO_RATIO) {
254 func = infoGainRatio;
255 firstBetter = greater(); // cur(first) > select(second)
256 } else /* if constexpr (Criterion == CriterionType::GINI_INDEX) */ {
257 func = giniIndex;
258 firstBetter = less(); // cur(first) < select(second)
259 }
260
261 vector<i32> attrIdxs(1, 0);
262 f64 cur;
263
264 auto totalWeight = getTotalWeight(examples);
265 auto availExamples = getAvailEx(attributes[0], examples);
266 auto normalizedWeight = getTotalWeight(availExamples) / totalWeight;
267 auto select = normalizedWeight * func(attributes[0], target, availExamples);
268
269 if constexpr (PrintInfo) {
270 cout << attributes[0]->name << ' ' << select << endl;
271 }
272 for (i32 i = 1, end = static_cast<i32>(attributes.size()); i < end; ++i) {
273 availExamples = getAvailEx(attributes[i], examples);
274 normalizedWeight = getTotalWeight(availExamples) / totalWeight;
275 cur = normalizedWeight * func(attributes[i], target, availExamples);
276 if constexpr (PrintInfo) {
277 cout << attributes[i]->name << ' ' << cur << endl;
278 }
279 if (firstBetter(cur, select)) {
280 attrIdxs.clear();
281 select = cur;
282 attrIdxs.push_back(i);
283 } else if (cur == select) {
284 attrIdxs.push_back(i);
285 }
286 }
287 if constexpr (PrintInfo) {
288 cout << endl;
289 }
290
291 if (nodeInfo) {
292 nodeInfo->first = getCriterionTypeStr(Criterion);
293 nodeInfo->second = select;
294 }
295 Attribute* dst = attributes[random::choose(attrIdxs)];
296
297 if constexpr (PrintInfo) {
298 cout << "choose " << dst->name << endl;
299 }
300 return dst;
301 }
