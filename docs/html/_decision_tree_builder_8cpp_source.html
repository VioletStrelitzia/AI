13 auto DecisionTreeBuilder::quickSort(vector<pair<Example*, f64>>& examples, Attribute* const& attr, i32 const& left, i32 const& right) -> void {
14 if (left < right) {
15 // 左闭右闭
16 i32 i = left + 1, j = right, p = left;
17 swap(examples[left], examples[(left + right) >> 1]);
18 while (true) {
19 while (examples[i].first->data[attr->index] <= examples[p].first->data[attr->index] && i < right) ++i;
20 while (examples[j].first->data[attr->index] >= examples[p].first->data[attr->index] && j > left) --j;
21 if (i < j) {
22 swap(examples[i], examples[j]);
23 } else {
24 break;
25 }
26 }
27 swap(examples[j], examples[p]);
28 quickSort(examples, attr, left, j);
29 quickSort(examples, attr, i, right);
30 }
31 }

109 auto DecisionTreeBuilder::splitExamples(
110 Attribute* const& attr,
111 vector<pair<Example*, f64>> const& examples
112 ) -> vector<vector<pair<Example*, f64>>> {
113 vector<vector<pair<Example*, f64>>> subExamples(attr->values.size());
114 switch (attr->type) {
115 case AttributeType::DISCRETE: {
116 i32 i = 0;
117 map<f64, i32> index;
118 for (f64 const& value: attr->values) {
119 index[value] = i;
120 ++i;
121 }
122 for (pair<Example*, f64> const& exp: examples) {
123 subExamples[index[exp.first->data[attr->index]]].push_back(exp);
124 }
125 } break;
126 case AttributeType::CONTINUOUS: {
127 subExamples.push_back(vector<pair<Example*, f64>>());
128 i32 i;
129 bool flag;
130 // 这里使用遍历匹配区间算法(nm)，而不是先排序再找区间端点(nlogn)，因为 n>>m，前者效率高
131 for (pair<Example*, f64> const& exp: examples) {
132 // 遍历各个区间，在区间内则加入
133 auto cit = attr->values.cbegin();
134 auto cnext = ++attr->values.cbegin();
135 auto cend = attr->values.cend();
136 i = 0, flag = false;
137
138 // 区间 (-, values[0]]
139 if (cit != cend && exp.first->data[attr->index] <= *cit) {
140 subExamples[i].push_back(exp);
141 continue;
142 }
143
144 // 区间 (values[i-1], values[i]]
145 ++i;
146 while (cnext != cend) {
147 if (*cit < exp.first->data[attr->index] && exp.first->data[attr->index] <= *cnext) {
148 subExamples[i].push_back(exp);
149 flag = true;
150 break;
151 }
152 ++cit, ++cnext, ++i;
153 }
154 if (flag) continue;
155
156 // 区间 (values[end], +]
157 if (cit != cend && exp.first->data[attr->index] > *cit) {
158 subExamples[i].push_back(exp);
159 }
160 }
161 } break;
162 }
163 return subExamples;
164 }

166 auto DecisionTreeBuilder::mergeExamples(vector<vector<pair<Example*, f64>>>& subExamples, vector<pair<Example*, f64>> const& absentExamples) -> void {
167 // 有效样本总权重
168 f64 n = 0., p;
169 vector<f64> n_k(subExamples.size(), 0.);
170 i32 i = 0;
171 for (vector<pair<Example*, f64>>& exs: subExamples) {
172 for (pair<Example*, f64> const& exp: exs) {
173 n_k[i] += exp.second;
174 n += exp.second;
175 }
176 ++i;
177 }
178 i = 0;
179 for (vector<pair<Example*, f64>>& exs: subExamples) {
180 // 依有效样本概率分布将缺失值样本划分到各个子集中
181 p = n_k[i] / n;
182 for (pair<Example*, f64> const& absExp: absentExamples) {
183 exs.push_back({ absExp.first, absExp.second * p });

199 auto DecisionTreeBuilder::pluralityValue(Attribute* const& target, vector<pair<Example*, f64>> const& examples) -> f64 {
200 unordered_map<f64, f64> counts;
201 vector<f64> targetV;
202
203 for (f64 const& targetValue: target->values) {
204 counts[targetValue] = 0;
205 }
206 for (auto const& p: examples) {
207 counts[p.first->target] += p.second;
208 }
209
210 targetV.push_back(*(target->values.cbegin()));
211 f64 max = counts[targetV.back()];
212 for (f64 const& targetValue: target->values) {
213 if (max < counts[targetValue]) {
214 max = counts[targetValue];
215 targetV.clear();
216 targetV.push_back(targetValue);
217 } else if (max == counts[targetValue]) {
218 targetV.push_back(targetValue);
219 }
220 }

261 auto DecisionTreeBuilder::infoGain(
262 Attribute* const& attr,
263 Attribute* const& target,
264 vector<pair<Example*, f64>>& examples) -> f64 {
265 f64 HL = 1., // 目标属性 target 在样本集的信息熵
266 HL_A = DBL_MAX; // 目标属性 target 在用属性 attr 分割样本集之后的加权信息熵，即条件信息熵
267
268 // 计算目标属性 target 在样本集的信息熵
269 HL = statistics::entropy(targetProbDist(target, examples));
270
271 switch (attr->type) {
272 case AttributeType::DISCRETE: {
273 // 如果属性 attr 是离散的，可以直接计算剩余期望熵
274 HL_A = entropyRemain(attr, target, splitExamples(attr, examples));
275 } break;
276 case AttributeType::CONTINUOUS: {
277 // 如果属性 attr 是离散的，需要枚举所有取值作为二分点的情况，并取剩余期望熵最小的点作为二分点
278 f64 temp, splitValue;
279 rg::sort(examples, [&attr](pair<Example*, f64> const& p1, pair<Example*, f64> const& p2) -> bool {
280 return p1.first->data[attr->index] < p2.first->data[attr->index];
281 });
282 //quickSort(examples, attr);
283 for (i32 i = 0, end = (i32)examples.size() - 1; i < end && examples[i + 1].first->data[attr->index] != Attribute::ABSENT; ++i) {
284 attr->values.clear();
285 attr->values.insert(examples[i].first->data[attr->index]);
286 temp = entropyRemain(attr, target, splitExamples(attr, examples));
287 if (HL_A > temp) {
288 splitValue = (examples[i].first->data[attr->index] + examples[i + 1].first->data[attr->index]) / 2;
289 HL_A = temp;
290 }
291 }
292 // 将二分点存到属性信息里
293 attr->values.clear();
294 attr->values.insert(splitValue);
295 } break;
296 }
297
298 return HL - HL_A;
299 }

301 auto DecisionTreeBuilder::infoGainRatio(
302 Attribute* const& attr,
303 Attribute* const& target,
304 vector<pair<Example*, f64>>& examples) -> f64 {
305 f64 n = getTotalWeight(examples),
306 H_A = 1.;
307
308 f64 HL = infoGain(attr, target, examples);
309
310 switch (attr->type) {
311 case AttributeType::DISCRETE: {
312 H_A = statistics::entropy(targetProbDist(target, examples));
313 } break;
314 case AttributeType::CONTINUOUS: {
315 vector<f64> P(attr->values.size() + 1, 0.);
316 i32 i;
317 bool flag;
318 for (pair<Example*, f64> const& exp: examples) {
319 auto cit = attr->values.cbegin();
320 auto cnext = ++attr->values.cbegin();
321 auto cend = attr->values.cend();
322 i = 0, flag = false;
323
324 if (cit != cend && exp.first->data[attr->index] <= *cit) {
325 ++P[i];
326 continue;
327 }
328
329 ++i;
330 while (cit != cend && cnext != cend) {
331 if (*cit < exp.first->data[attr->index] && exp.first->data[attr->index] <= *cnext) {
332 ++P[i];
333 flag = true;
334 break;
335 }
336 ++cit, ++cnext, ++i;
337 }
338 if (flag) continue;
339
340 if (cit != cend && exp.first->data[attr->index] > *cit) {
341 ++P[i];
342 }
343 }
344 for (f64& p: P) p /= n;
345 H_A = statistics::entropy(P);
346 } break;
347 }
348 if (H_A == 0) {
349 return 0;
350 }
351 return HL / H_A;
352 }
