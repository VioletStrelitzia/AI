171 void State::zeroMoveTo(Vector2 const& swapPos) {
172 // 单线程环境下使用静态变量减少生成临时变量反复申请内存
173 // 多线程环境下就不要用静态变量以防数据竞争
174 static usize tmp1, tmp2;
175 static unsigned idx1, idx2, offset;
176
177 tmp1 = data & mask[zeroPos.r][zeroPos.c];
178 tmp2 = data & mask[swapPos.r][swapPos.c];
179 data -= tmp1;
180 data -= tmp2;
181 idx1 = zeroPos.r * rows + zeroPos.c;
182 idx2 = swapPos.r * rows + swapPos.c;
183 if (idx1 > idx2) {
184 offset = (idx1 - idx2) << 2;
185 tmp1 >>= offset;
186 tmp2 <<= offset;
187 }
188 else {
189 offset = (idx2 - idx1) << 2;
190 tmp1 <<= offset;
191 tmp2 >>= offset;
192 }
193 data ^= tmp1;
194 data ^= tmp2;
195
196 zeroPos = swapPos;
197 }

199 void State::updateNextActions(Action const* const& lastAction) {
200 nextActions.clear();
201 if (zeroPos.r > 0 && lastAction != actions + 1) nextActions.push_back(actions);
202 if (zeroPos.r < rows - 1 && lastAction != actions) nextActions.push_back(actions + 1);
203 if (zeroPos.c > 0 && lastAction != actions + 3) nextActions.push_back(actions + 2);
204 if (zeroPos.c < cols - 1 && lastAction != actions + 2) nextActions.push_back(actions + 3);

230 void State::genPrecision(State const& originalNFPState) {
231 ofstream outFile(precisionFilename, std::ios::binary);
232 if (outFile.good()) {
233 outFile.write(reinterpret_cast<char const*>(&rows), sizeof rows);
234 outFile.write(reinterpret_cast<char const*>(&cols), sizeof cols);
235 outFile.write(reinterpret_cast<char const*>(&maxDepth), sizeof maxDepth);
236
237 usize m = 0b1111;
238 for (int r = 0; r < rows; ++r) {
239 for (int c = 0; c < cols; ++c) {
240 mask[r][c] = m;
241 outFile.write(reinterpret_cast<char*>(& m), sizeof m);
242 m <<= 4;
243 }
244 }
245
246 allNFPStates.clear();
247
248 allNFPStates.push_back(_targetNFPState);
249 outFile.write(reinterpret_cast<char*>(&_targetNFPState), 16);
250 depthIndexs[0] = 1;
251 set<usize> s;
252
253 bool flag = false;
254 int index = 0;
255 while (index < allNFPStates.size()) {
256 s.insert(allNFPStates[index].data);
257
258 for (Action const* const& action : allNFPStates[index].nextActions) {
259 State nextNFPState(allNFPStates[index]);
260 nextNFPState.executeAction(action);
261 if (nextNFPState.depth > maxDepth) {
262 flag = true;
263 break;
264 }
265 if (s.find(nextNFPState.data) == s.end()) {
266 outFile.write(reinterpret_cast<char*>(&nextNFPState), 16);
267 allNFPStates.emplace_back(nextNFPState);
268 depthIndexs[nextNFPState.depth] = static_cast<int>(allNFPStates.size());
269 }
270 }
271
272 if (flag) {
273 break;
274 }
275 ++index;
276 }
277 }
278 unsigned zero[4] = { 0 };
279 outFile.write(reinterpret_cast<char*>(zero), 16);
280 outFile.write(reinterpret_cast<char*>(&depthIndexs), sizeof depthIndexs);
281 for (int i = 0; i <= maxDepth; ++i) {
282 cout << depthIndexs[i] << endl;
283 }
284 cout << "allNFPStates.size() = " << allNFPStates.size() << endl;
285
286 outFile.close();
