84 auto classify(map<string, f64>& sample) const -> f64 override {
85 auto valueIt = sample.find(attribute);

99 auto dot(ofstream& outFile, i32& idx, string const& fromParent) const -> void override {
100 string color = "#e58139";
101 if (lchild && rchild) {
102 color = "#64a1c3";
103 } else if (!lchild && !rchild) {
104 color = "#451247";

106 outFile << format(

108 idx, fromParent, sampleCount, attribute, splitInfo.first, splitInfo.second, classification, color);
109 i32 curIdx = idx;
110 if (lchild) {
111 ++idx;

167 idx, fromParent, sampleCount, attribute, splitInfo.first, splitInfo.second, color);
168 i32 curIdx = idx;
169 for (auto const& child: children) {
170 if (child.second != nullptr) {
171 ++idx;
172 outFile << format("{} -> {};\n", curIdx, idx);
173 child.second->dot(outFile, idx, format("{} == {}", attribute, child.first));
174 }
175 }
176 }
