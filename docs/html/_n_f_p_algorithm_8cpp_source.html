97 auto Algorithm::IDS(State& originalNFPState, State const& targetNFPState, Vector2 const& preZeroPos, i32 const& depthLimit, i32& ans) -> bool {
98 if (originalNFPState == targetNFPState) {
99 ans = originalNFPState.depth;
100 return true;
101 }
102
103 // 当前步数（深度） + 估价 > 深度限制，立即回溯
104 if (manhattanDistance(originalNFPState, targetNFPState) > depthLimit) {
105 return false;
106 }
107
108 // 枚举可以执行的action，即每次向上、下、左、右移动
109 for (i32 actionId = 0; actionId < State::actionsNum; ++actionId) {
110 Vector2 zeroPos = originalNFPState.zeroPos; // 0 的位置
111 Vector2 swapPos = zeroPos + State::actions[actionId]; // 要交换的数 的位置
112
113 // 优化：如果 0 将要回到回到上一次的位置，会判断直接跳过这次 action 而不往下搜索，避免提前回溯重复搜索
114 // 以及越界判断
115 if (preZeroPos == swapPos ||
116 swapPos.r < 0 || swapPos.r >= State::rows ||
117 swapPos.c < 0 || swapPos.c >= State::cols) {
118 continue;
119 }
120
121 // 交换 0 和 交换的数码 的位置
122 originalNFPState.zeroMoveTo(swapPos);
123 ++originalNFPState.depth;
124
125 if (underExperiment) {
126 ++nodeCountTemp;
127 }
128 if (IDS(originalNFPState, targetNFPState, swapPos, depthLimit, ans)) {
129 // 回溯的过程中顺便输出步骤
130 if (printPath) {
131 cout << "第" << originalNFPState.depth << "步:" << State::actionInfo[actionId] << endl;// << originalNFPState << endl << endl;
132 }
133
134 originalNFPState.zeroMoveTo(zeroPos);
135 --originalNFPState.depth;
136 return true;
137 }
138
139 // 回溯，交换回来 空格0 和 旁边要交换的数 的位置，并回溯 originalNFPState 零的位置，预备下一个 action 的搜索
140 originalNFPState.zeroMoveTo(zeroPos);
141 --originalNFPState.depth;
142 }
143 return false;
144 }

209 i32 Algorithm::DLS(State& originalNFPState, State const& targetNFPState, i32 const& depthLimit) {
210 originalNFPState.reset();
211
212 stack<State> s;
213 StateSet set;
214 s.push(originalNFPState);
215 if (underExperiment) {
216 ++nodeCountTemp;
217 }
218 while (!s.empty()) {
219 if (s.top() == targetNFPState) {
220 if (printPath) {
221 s.top().printPath();
222 }
223 return s.top().depth;
224 }
225 set.visit(s.top());
226 bool flag = true;
227 if (s.top().depth < depthLimit) {
228 while (s.top().hasNextAction()) {
229 State nextNFPState(s.top().genNextNFPState());
230 if (!set.visited(nextNFPState)) {
231 s.emplace(nextNFPState);
232 if (underExperiment) {
233 ++nodeCountTemp;
234 }
235 flag = false;
236 break;
237 }
238 }
239 }
240 if (flag) {
241 set.remove(s.top());
242 s.pop();
243 }
244 }

249 i32 Algorithm::incorrectCount(State const& originalNFPState, State const& targetNFPState) {
250 i32 incorrectCount = 0;
251 static Vector2 pos[2][State::size];
252
253 for (i32 r = 0; r < State::rows; ++r) {
254 for (i32 c = 0; c < State::cols; ++c) {
255 pos[0][originalNFPState.at(r, c)] = { r, c };
256 pos[1][targetNFPState.at(r, c)] = { r, c };
257 }
258 }
259
260 for (i32 i = 1; i < State::size; ++i) {
261 if (pos[0][i] != pos[1][i]) {
262 ++incorrectCount;
263 }
264 }
265
266 return incorrectCount + originalNFPState.depth;
267 }

269 i32 Algorithm::manhattanDistance(State const& originalNFPState, State const& targetNFPState) {
270 i32 manhattanDis = 0;
271 static Vector2 pos[2][State::size];
272
273 for (i32 r = 0, i = 0; r < State::rows; ++r) {
274 for (i32 c = 0; c < State::cols; ++c, i += 4) {
275 pos[0][originalNFPState.at(r, c)] = { r, c };
276 pos[1][targetNFPState.at(r, c)] = { r, c };
277 }
278 }
279
280 for (i32 i = 1; i < State::size; ++i) {
281 manhattanDis += abs(pos[0][i].r - pos[1][i].r) + abs(pos[0][i].c - pos[1][i].c);
282 }
283
284 return manhattanDis + originalNFPState.depth;
285 }

288 originalNFPState.reset();
289
290 priority_queue<pair<i32, State>, vector<pair<i32, State>>, std::greater<pair<i32, State>>> pq;
291 StateSet set;
292 pq.push({ heuristicFunc(originalNFPState, targetNFPState), originalNFPState });
293 if (underExperiment) {
294 ++nodeCountTemp;
295 }
296 while (!pq.empty()) {

309 pq.emplace(heuristicFunc(nextNFPState, targetNFPState), nextNFPState);
