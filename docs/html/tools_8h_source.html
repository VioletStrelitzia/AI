315 auto median(vector<ElemType>& data, usize const& left, usize const& right, usize const& mid,
316 function<bool(ElemType const&, ElemType const&)> const& comper) -> ElemType {
317 usize i = left + 1, j = right, p = left;
318 swap(data[left], data[random::uniformInt<usize>(left, right)]);
319 while (true) {
320 while (i < right && comper(data[p], data[i])) ++i;
321 while (j > left && !comper(data[p], data[j])) --j;
322 if (i < j) {
323 swap(data[i], data[j]);
324 } else {
325 break;
326 }
327 }
328 if (j == mid) {
329 return data[p];
330 }
331 swap(data[j], data[p]);
332 if (j < mid) {
333 return median(data, i, right, mid, comper);
334 }
335 return median(data, left, j, mid, comper);
336 }

338 template <typename ElemType>
339 auto median(vector<ElemType>& data, usize left, usize right,
340 function<bool(ElemType const&, ElemType const&)> const& comper) -> ElemType {
341 usize i, j, p, mid = (left + right + 1) >> 1;
342 while (true) {
343 i = left + 1, j = right, p = left;
344 swap(data[left], data[random::uniformInt<usize>(left, right)]);
345 while (true) {
346 while (i < right && comper(data[p], data[i])) ++i;
347 while (j > left && !comper(data[p], data[j])) --j;
348 if (i < j) {
349 swap(data[i], data[j]);
350 } else {
351 break;
352 }
353 }
354 if (j == mid) {
355 return data[p];
356 }
357 swap(data[j], data[p]);
358 if (j < mid) {
359 left = i;
360 } else {
361 right = j;
362 }
363 }

371 template <typename T>
372 auto entropy(unordered_map<T, f64> const& PMap) -> f64 {
373 f64 H = 0.;
374 for (pair<T, f64> const& p: PMap) {
375 if (p.second == 0.) {
376 continue;
377 }
378 H -= p.second * log2(p.second);
379 }
380 return H;
381 }

383 template <typename T>
384 auto gini(unordered_map<T, f64> const& PMap, f64 const& n) -> f64 {
385 f64 G = 1.;
386 f64 n2 = n * n;
387 for (pair<T, f64> const& p: PMap) {
388 G -= p.second * p.second / n2;
389 }
390 return G;
391 }

393 template <typename T>
394 auto gini(unordered_map<T, pair<f64, f64>> const& PMap, f64 const& n) -> pair<f64, f64> {
395 pair<f64, f64> G = { 1., 1. };
396 f64 n2 = n * n;
397 for (pair<T, pair<f64, f64>> const& p: PMap) {
398 G.first -= p.second.first * p.second.first / n2;
399 G.second -= p.second.second * p.second.second / n2;
400 }
401 return G;
