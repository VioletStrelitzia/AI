491template <typename T>
492auto entropy(unordered_map<T, f64> const& PMap) -> f64 {
493 f64 H = 0.;
494 for (pair<T, f64> const& p: PMap) {
495 if (p.second == 0.) {
496 continue;
497 }
498 H -= p.second * log2(p.second);
499 }
500 return H;
501}

503template <typename T>
504auto gini(unordered_map<T, f64> const& PMap, f64 const& n) -> f64 {
505 f64 G = 1.;
506 f64 n2 = n * n;
507 for (pair<T, f64> const& p: PMap) {
508 G -= p.second * p.second / n2;
509 }
510 return G;
511}
512
513template <typename T>
514auto gini(unordered_map<T, pair<f64, f64>> const& PMap, f64 const& n) -> pair<f64, f64> {
515 pair<f64, f64> G = { 1., 1. };
516 f64 n2 = n * n;
517 for (pair<T, pair<f64, f64>> const& p: PMap) {
518 G.first -= p.second.first * p.second.first / n2;
519 G.second -= p.second.second * p.second.second / n2;
520 }
521 return G;
522}
