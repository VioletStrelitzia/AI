41auto variance(vector<f64> const& data) -> f64 {
42 f64 prevMean = 0.,
43 curMean = 0.,
44 s = 0.;
45 usize n = data.size();
46
47 for (usize i = 0; i < n; ++i) {
48 prevMean = curMean;
49 curMean += (data[i] - curMean) / (i + 1);
50 s += (data[i] - curMean) * (data[i] - prevMean);
51 }
